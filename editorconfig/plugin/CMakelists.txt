# New escape-sequence processing.
cmake_minimum_required(VERSION 3.16.3)
cmake_policy(VERSION 3.16.3)

project(editorconfig-plugin-tests NONE)

if(${CMAKE_SOURCE_DIR} STREQUAL ${CMAKE_CURRENT_SOURCE_DIR})
    set(GETOPTIONS_CMD ${CMAKE_CURRENT_LIST_DIR}/../../bin/getoptions CACHE STRING "getoptions command.")
    enable_testing()
    message(STATUS "tests: Standalone testing enabled")
endif()

function(add_getoptions_test name buffer_path options regex)
    set(TEST_NAME ec_plugin_${name})
    add_test(NAME ${TEST_NAME} COMMAND ${GETOPTIONS_CMD}
        -n "${CMAKE_CURRENT_SOURCE_DIR}/files/test_files/${buffer_path}"
        -c "let yaifa_use_heuristics=0"
        "${options}")
    set_tests_properties(${TEST_NAME} PROPERTIES PASS_REGULAR_EXPRESSION "${regex}")
endfunction()
